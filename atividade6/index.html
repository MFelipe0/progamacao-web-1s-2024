const express = require('express'); const path = require('path'); const app = express(); const PORT = 3000; app.use(express.urlencoded({ extended: true })); app.set('views', path.join(__dirname, 'views')); app.set('view engine', 'html'); app.engine('html', require('ejs').renderFile); app.get('/', (req, res) => { res.render('index'); }); app.post('/dados', (req, res) => { const { nome, endereco, telefone, data } = req.body; res.render('dados', { nome, endereco, telefone, data }); }); app.listen(PORT, () => { console.log(`Servidor rodando na porta ${PORT}`); });
